function get_varTAS(X)

% This might be needed
clear ncinfo ncreadatt ncwriteatt ncread
close all

% Load time vector to reference length
try
    Time=ncread(X.RawPath,'time');
    rawTimeVar = 'time';
catch
    Time = ncread(X.RawPath,'Time');
    rawTimeVar = 'Time';
end

% Make sure ncFINAL exists
try
    test=ncinfo(X.ncFINAL);
catch
    error('get_varTAS:  X.ncFINAL does not exist');
end

% Variables to be used in TAS etc.
PressUsed = X.PressUsed;
TempUsed = X.TempUsed;
PitotUsed =  X.PitotUsed;

TT=datetime('now');
% Physical constants structure
C=phycon;
Tzero=C.Tzero;

% Get raw variables needed for tas group
%
orate=X.procRate;
Rate=orate;
rawfile=X.RawPath;

%  Need TAS and temp and pressure groups
%  See column names on Ptable structure
P = X.Ptable;
colNames = P.Properties.VariableNames;
wantNames={'TAS', 'HADS', 'CPT', 'WESTON'};
mask = false(size(colNames));
for p = wantNames
    mask = mask | contains(colNames, p{1});
end
GROUPS = colNames(mask);
% Use pivot table and raw mapping table to get
% 1. variables to be calculated;
% 2. raw measurements needed from *_raw.nc
[arcNames, rawNames] = getVarsAndRawNames(X.Ptable, GROUPS, X.Ttable);

if ismember(GROUPS,'HADS')
    matTAS=fullfile(X.tempdir,sprintf("%s_HADS.mat", X.BaseName));
    load(matTAS,'ps_hads_a','ps_hads_b');
end
if ismember(GROUPS,'CPT')
    matTAS=fullfile(X.tempdir,sprintf("%s_CPT.mat", X.BaseName));
    load(matTAS,'CPT6140','CPT9000','TEMP9000');
end

% Get the necessary variables from the _raw.nc file; change rate
rawfile = X.RawPath;
for ii=1:length(rawNames)
    var1=rawNames(ii);
    ss=sprintf("blurf=ncread(rawfile,'%s');",var1);
    try
        eval(ss)
    catch
        {var1 'not found'}
        continue
    end
    [irate,dims,frate]=get_irate(rawfile,var1,orate);
    if(irate>1)
        [ninterp,ndecim]=interp_decim(irate,orate);
        j=find(~isnan(blurf));
        blurf=blurf(:);
        blurf=interp1(j,blurf(j),[1:numel(blurf)],[],0);
        blurf=changeRate(Ninterp(blurf,ninterp),ndecim,'FIR',X.FillValue);
    else
        ninterp=1;
        ndecim=1;
    end
    ss=sprintf("%s=blurf(:);",var1);eval(ss);
end

%  See which are available
Pressures = {};
% HADSA
if exist('HADSA_PER','var') & ~isempty(HADSA_PER)
    ps_hads_a=get_hads(X,'A',HADSA_VR,HADSA_VT,HADSA_PER);
    Pressures{end+1} = {'ps_hads_a'};
else
    ps_hads_a = zeros(size(BIAS));
end
% HADSB
if exist('HADSB_PER','var') & ~isempty(HADSB_PER)
    ps_hads_b=get_hads(X,'B',HADSB_VR,HADSB_VT,HADSB_PER);
    Pressures{end+1} = {'ps_hads_b'};
else
    ps_hads_b = zeros(size(BIAS));
end
% WESTON WDPM
if exist('WDPM','var') & ~isempty(WDPM)
    ps_weston=WDPM;
    Pressures{end+1} = {'ps_weston'};
else
    ps_weston = zeros(size(BIAS));
end
% CPT 6140
if exist('PRES6140','var') & ~isempty(PRES6140)
    ps_CPT6140=PRES6140;
    Pressures{end+1} = {'ps_CPT6140'};
else
    ps_CPT6140 = zeros(size(BIAS));
end
% CPT 9000 pressure
if exist('PRES9000','var') & ~isempty(PRES9000)
    ps_CPT9000=PRES9000;
    Pressures{end+1} = {'ps_CPT9000'};
else
    ps_CPT9000 = zeros(size(BIAS));
end
% CPT 9000 temperature
if exist('TEMP9000','var') & ~isempty(TEMP9000)
    temp_CPT9000=TEMP9000;
else
    temp_CPT9000 = zeros(size(BIAS));
end

% Which should be used (call it PSM)
varUsed=[];
Pressures = string(Pressures);
jj = find(matches(Pressures,PressUsed));
ss = sprintf("PSM = %s;",Pressures{jj});
eval(ss);

% Indicated airspeeds [knots]
aias=q2ias(AIAS);
bias=q2ias(BIAS);

% humidity (TDPEDGE or TDPEGG (EG&G)dbcont
if(cell_find(arcNames,'tdp'))
    try
        tdp=TDPEDGE; % in centigrade
        ncwriteatt(X.ncFINAL,'tdp','Sensor','Edgetech');
    catch
        try % OK, try this
            blurf=ncread(X.RawPath,'TDP');TDP=blurf(:);
            tdp=TDP; %  EG&G
            ncwriteatt(X.ncFINAL,'tdp','Sensor','EG&G');
        catch
            "getvarTAS: No depoint found"
            return
        end
    end
    kk=find(tdp<0);
    tdp0=tdp;
    tdp1=tdp;
    % Assume Frost Point sensed if tdp<0 C
    if(~isempty(kk))
        tdp(kk) = dew(tdp(kk)); % Frost point is sensed
        % tdp1(kk) = frostpoint_to_dewpoint(tdp1(kk));
    end
    TDPK=tdp+Tzero;
end

% Air speed vector and pressure correction
% get flow angles and static pressure correction
%
if(mean(DPA)<0); %Sanity check
    DPA=-DPA;
end
DPB=-DPB; % Always

% Attitude angles and pressure correction (Sys09 version)
% addpath(fullfile(X.Source,'mfiles858','mfiles858_0'));
% These are from Sys09
rmpath(fullfile(X.Repo,'Sys26','Old-files-keep-for-now'));
addpath(fullfile(X.Repo,'Sys26','Old-files-keep-for-now/mfiles858_0'));
kk = find( ~isnan(AIAS) & ~isinf(AIAS) & AIAS>0);
AIAS = interp1(kk,AIAS(kk),[1:numel(AIAS)]','spline',0);
kk = find( ~isnan(DPR) & ~isinf(DPR) & DPR>0);
DPR = interp1(kk,DPR(kk),[1:numel(DPR)]','spline',0);
kk = find( ~isnan(TRF+Tzero) & ~isinf(TRF+Tzero) & TRF+Tzero>0);
TRF = interp1(kk,TRF,[1:numel(TRF)]','spline',0);

TA = tanAlpha(DPA,DPB,DPR);
TB = tanBeta(DPB,DPR);
alpha=atan(TA); % radians
beta=atan(TB);
%  Matlab has function alpha and beta.  Next is a workaround
alphaa = alpha;
betaa = beta;
% Output variables
dpb=DPB;
dpa=DPA;
dpr=DPR;

% R858 flow angle calibration factors
[pitoff,rolloff,hedoff,bfactor,afactor,Qfactor] = AV410_factors(X.ncFINAL);
%
% Mixing Ratio:
mr=0;
vaporPressure=sat_vapor_pressure_GG(TDPK); % Use Goff-Gratch
%  Iteration needed for higher humidity levels
%     since mr needed for boom_pcor
for jj=1:3
    boom_pcor = cone_pcor(AIAS,DPB,DPA,DPR,PSM,mr);
    mr=(C.Mv./C.Md).*vaporPressure./(PSM - boom_pcor - vaporPressure); % g/g
    mean(mr)
end
[boom_pcor,qc0,fcoef] = cone_pcor(AIAS,DPB,DPA,DPR,PSM,mr);
q_impact = qc0 * (1+Qfactor);  % This is a residual residual static error

% Apply pressure corrections
for jj=1:numel(Pressures);
    if(exist(Pressures{jj}))
        ss=sprintf("%s = %s - boom_pcor;",Pressures{jj},Pressures{jj});
        eval(ss)
    end
end

% Compute moist-air thermodynamic and airspeed quantities using measured:
%   - Ps_meas  : Measured static pressure (hPa)PSX
%   - Pt       : Pitot (total) pressure (hPa)
%   - Tm       : Measured probe temperature (K)
%   - r        : Probe recovery factor (dimensionless, required)
%   - Td       : Dewpoint temperature (K)
% Static-pressure correction (name,value pair)
%   "dPs_corr", pcor
% q_impact is impact (stagnation) pressure
% q_dyn = dynamic pressure = 1/2*rho*V^2;
% q_dyn = q_impact if incompressible.
PSX = PSM - boom_pcor;

% Total pressure
%  Pt = q_impact + Ps;
pTotal = q_impact + PSX; %

clear TASX
% Get moist air properties and airspeeds
if exist("TRF")
    %recovery factor
    r=0.6425;
    Tmeas = TRF + Tzero; % Uncorrected
    ADatTRF = airdata(PSM, pTotal, Tmeas, r, TDPK, "dPs_corr",boom_pcor);
    trf = ADatTRF.Ts;
    if X.TempUsed == 'trf'
        ADat = ADatTRF;
        TEMPX = trf;
        TASX = ADat.TAS;
    end
    units=ncreadatt(X.ncFINAL,'trf','units');
    trf=convertUnits(trf,ADat.units.Ts,units);
end
if exist("TROSE")
    %recovery factor
    r=0.97;
    Tmeas = TROSE + Tzero; % Uncorrected
    ADatTROSE = airdata(PSM, pTotal, Tmeas, r, TDPK, "dPs_corr",boom_pcor);
    trose = ADatTROSE.Ts;
    if X.TempUsed == 'trose'
        ADat = ADatTROSE;
        TEMPX = trose;
        TASX = ADat.TAS;
    end
    units=ncreadatt(X.ncFINAL,'trose','units');
    trose=convertUnits(trose,ADat.units.Ts,units);
end

% Sanity check
if ~exist("TASX")
    error("get_varTAS:  No TempUsed variable found")
end

units=ncreadatt(X.ncFINAL,'tas','units');
tas = convertUnits(TASX,ADat.units.TAS,units);

% Pressure altitude
PALT = ADat.Zp;
    units=ncreadatt(X.ncFINAL,'PALT','units');
    PALT = convertUnits(PALT,ADat.units.Zp,units);
% Relative Humidity
    rh = ADat.RH; % fraction
    units=ncreadatt(X.ncFINAL,'rh','units');
    if units == 'percent' & ADat.units.RH == "1"
        rh = rh.*100;
    end
% Mixing Ratio: leave as kg/kg  for  now
    mr = ADat.w; % kg/kg
    units=ncreadatt(X.ncFINAL,'mr','units');
    mr = convertUnits(mr,ADat.units.w,'kg/kg');
% Dry theta
    thetad = ADat.thetad;
    units=ncreadatt(X.ncFINAL,'thetad','units');
    thetad = convertUnits(thetad,ADat.units.thetad,'Kelvin');
% Thetae
    thetae = ADat.theta_e;
    units=ncreadatt(X.ncFINAL,'thetae','units');
    thetae = convertUnits(thetae,ADat.units.theta_e,'Kelvin');
% Lifted Condensation Level LCL temp and pressure
    TLCL = ADat.TLCL;
    units=ncreadatt(X.ncFINAL,'TLCL','units');
    TLCL = convertUnits(TLCL,ADat.units.TLCL,'Kelvin');
    PLCL = ADat.PLCL;
    units=ncreadatt(X.ncFINAL,'PLCL','units');
    PLCL = convertUnits(PLCL,ADat.units.PLCL,'hPa');

ncwriteatt(X.ncFINAL,'tas','TempUsed',X.TempUsed,'Datatype','char');
ncwriteatt(X.ncFINAL,'tas','PressUsed',X.PressUsed,'Datatype','char');
ncwriteatt(X.ncFINAL,'TASX','TempUsed',X.TempUsed,'Datatype','char');
ncwriteatt(X.ncFINAL,'TASX','PressUsed',X.PressUsed,'Datatype','char');
ncwriteatt(X.ncFINAL,'thetad','PressUsed',X.PressUsed,'Datatype','char');
ncwriteatt(X.ncFINAL,'thetae','TempUsed',X.TempUsed,'Datatype','char');
ncwriteatt(X.ncFINAL,'thetae','PressUsed',X.PressUsed,'Datatype','char');

% Eddy Dissipation Rate from TAS
%
% Get TAS components @ 50 Hz from raw.nc file
fs50 = 50;
fny=fs50/2;
f1 = 10;
f2 = 25;

%  Estimate Eddy Dissipation Rate ^ 1/3 power at 1 Hz;
[tasx,tasy,tasz] = get_uvw(X,fs50); % aircraft-relative TAS components
Tas = rssq([tasx,tasy,tasz],2); % root sum of squares
[epsiux,epsivy,epsiwi,varhatu,varhatv,varhatw,uwf,vwf] = ...
    EddyDissipationRate(Tas,tasx,tasy,tasz,f1,f2,fs50);
edrTAS=epsiwi.^.333;

%%%% New boom measurements
    randz = randn(size(TASX));
    dp1New858 = randz;
    dp2New858 = randz;
    psaNew858 = randz;
    psbNew858 = randz;
    ptbNew858 = pTotal;
    dpnNew858 = dpra_sim(fcoef,q_impact,TA,TB);
%%%%%%%%%%%%%%%%

Rate=X.procRate;Sps_hads_a=sprintf('sps%i',Rate);
time=ncread(X.RawPath,rawTimeVar);
xlsfile=fullfile(X.Source,X.PROJ,'../config/defines3.xlsx');

Rate=X.procRate;
SPS=sprintf('sps%i',Rate);
time=ncread(X.RawPath,rawTimeVar);

[irate,dims,frate]=get_irate(X.ncFINAL,'tas',X.procRate);

ss1="'orate','Rate','rawfile','arcNames','rawNames','alphaa','betaa','TEMPX','PSX','Time','ss1'";
for ii=1:numel(arcNames);
    if(numel(arcNames{ii})>1)
        ss1=sprintf("%s,'%s'",ss1,arcNames{ii});
    end
end

% Write variables out to matfile
matfile=fullfile(X.tempdir,sprintf("%s_TAS.mat",X.BaseName));
delete(matfile)

ss=sprintf("save(matfile,%s);",ss1);
eval(ss);

TT1=datetime('now');
procSeconds=seconds(TT1-TT)
save(matfile,'-append','procSeconds','TT','TT1')
;
load_ncFINAL(X.ncFINAL,matfile);
sprintf('Processed get_varTAS.m for Project: %s',X.PROJ)

end

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
%
% Calculate TAS components in aircraft coords at 50 Hz
%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
function [compx,compy,compz] = get_uvw(X,rate)

RawPath=X.RawPath;
TempUsed = X.TempUsed;
PressUsed = X.PressUsed;
% 1000 Hz variable
[TEMPK] = getdata(RawPath,upper(TempUsed),rate)+273.15;
[DPA] = getdata(RawPath,'DPA',rate);
if(mean(DPA)<0)
    DPA=-DPA;
end
[DPB] = getdata(RawPath,'DPB',rate); %reversed
DPB = - DPB; % always
[DPR] = getdata(RawPath,'DPR',rate);
[AIAS] = getdata(RawPath,'AIAS',rate);
switch PressUsed
    case 'ps_hads_a'
        HADSA_VR  = getdata(RawPath,'HADSA_VR',rate);
        HADSA_VT  = getdata(RawPath,'HADSA_VT',rate);
        HADSA_PER = getdata(RawPath,'HADSA_PER',rate);
        PMB = get_hads(X,'A',HADSA_VR,HADSA_VT,HADSA_PER);

    case 'ps_hads_b'
        HADSB_VR  = getdata(RawPath,'HADSB_VR',rate);
        HADSB_VT  = getdata(RawPath,'HADSB_VT',rate);
        HADSB_PER = getdata(RawPath,'HADSB_PER',rate);
        PMB = get_hads(X,'B',HADSB_VR,HADSB_VT,HADSB_PER);

    case 'WDPM'
        PMB = getdata(RawPath,'WDPM',rate);

    case 'ps_CPT6140'
        PMB = getdata(RawPath,'PRES6140',rate);

    otherwise
        'blurf'
end

ii=find(PMB>=300 | ~isnan(PMB));
PMB = interp1(ii,PMB(ii),1:numel(PMB),'spline',300)';

[pcorc,q_impact,tb,ta] = cone_pcor(AIAS,DPB,DPA,DPR,PMB);
ii=find(q_impact> 50 |  ~isnan(q_impact));
q_impact = interp1(ii,q_impact(ii),1:numel(q_impact),'spline',10)';

[TasHR]=tasf(q_impact,PMB,TEMPK); % m/s
ii=find(TasHR>=50 | ~isnan(TasHR));
TasHR = interp1(ii,TasHR(ii),1:numel(TasHR),'spline',10)';

va=tas2va(TasHR,atan(ta),atan(tb));

compx = va(:,1);
compy = va(:,2);
compz = va(:,3);

end
